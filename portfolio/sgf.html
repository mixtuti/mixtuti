サウンドガンファイト

現在も鋭意制作中の対戦型リズムゲーム。


ゲーム名 サウンドガンファイト
ジャンル 対戦型リズムゲーム
制作期間 制作中(2022年5月〜)
制作人数 4人
担当箇所 SE
音符の発射・判定、シーン遷移、ターン判定
使用ツール Unity、Cubase 12 Pro、DSK Saxophones
使用言語 C#
リリース日 2023年3月予定

このゲームは2022年に開催された「GW企画＆作業ジャム2022」にて制作した企画書をもとに、現在も制作しているゲームです。メンバーは4人です。

・概要

対戦型のリズムゲームで相手が発射した音符が、自分のところのラインを超えないように音符を打ち返す。というゲームです。

このゲームは、BitSummitに展示するために制作していたゲームです。しかし、残念ながら学内の選考に落ちてしまい展示は叶いませんでした。なので、今はUnityroomに投稿するために改良を重ねています。

・担当した箇所のソースコード

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;

public class DeleteObject : MonoBehaviour {
    public Score ScoreSclipt;
    string Judge = "Bad";
    bool time = false;
    public GameObject particleObject;

    void Start() { //0.1秒待つ処理
        Invoke(nameof(Stay), 0.1f);
    }

    bool Stay() {
        time = true;
        return time;
    }

    void OnCollisionStay(Collision other) { //何かに触れている時の処理
        if (other.gameObject.tag == "speaker") { //相手側のスピーカーに触れた時の処理
            if (Stay() == true) {
                Destroy(gameObject);
                Judge = "Miss";
                Debug.Log("1PMiss");
            }
        }
        else if (other.gameObject.tag == "2Pspeaker") { //相手側のスピーカーに触れた時の処理
            if (Stay() == true) {
                Destroy(gameObject);
                Judge = "Miss";
                Debug.Log("2PMiss");
            }
        }
    }

    void OnDestroy() { //消された時の処理
        if (Judge == "Good") {
            ScoreSclipt.Score1P(1000);
        }
        if (Judge == "Bad") {
            ScoreSclipt.Score1P(500);
        }
        Instantiate(particleObject, this.transform.position, Quaternion.identity);
        Debug.Log(Judge);
    }

    void OnTriggerEnter(Collider other) {
        if (other.gameObject.tag == "Line") {
            Judge = "Good";
        }
    }

    void OnTriggerExit(Collider other) {
        if (other.gameObject.tag == "Line") {
            Judge = "Bad";
        }
    }
}

DeleteObjectのスクリプトは、プレイヤーが発射するオブジェクトにアタッチして使います。相手が処理できずにスピーカーに当たった場合にミスの判定を出し、消された時は、パーティクルを呼び出し、消滅します。

今見てみると、同じような処理があるから、関数にしてまとめても良かったなと思っています。

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;
using UnityEngine.SceneManagement;

using AudioManager;

public class LoadingScene : MonoBehaviour
{
    //　非同期動作で使用するAsyncOperation
    private AsyncOperation async;

    [Header("ロード中に表示する画面"),SerializeField] private GameObject loadUI;//　シーンロード中に表示するUI画面 
    [Header("進捗バー"),SerializeField] private Slider slider; //　読み込み率を表示するスライダー
    [Header("次のシーン")] public SceneReference m_scene;
    [Header("次のシーン削除するオブジェクト"),SerializeField] List<GameObject> list = new List<GameObject>();

    void Start()
    {
        for (int i = 0; i < list.Count; i++) { 
            list[i].SetActive(true); 
        } 
    } 
        
    void Update() {
        if(SceneManager.GetActiveScene().name !="LoadScene" || SceneManager.GetActiveScene().name !="Result" ){
            if(Input.GetKeyDown("joystick button 7")){ 
                NextScene(); 
            } 
        } 
        else{ 
            if(Input.GetKeyDown("joystick button 2")){
                NextScene(); 
            } 
        } 
    } 
        
    public void NextScene() { //　それぞれの画面UIをアクティブにする 
        loadUI.SetActive(true);
        slider.gameObject.SetActive(true); 
        if (SceneManager.GetActiveScene().name=="LoadScene" ) {
            SEManager.Instance.Play(SEPath.SGF_MSE_DECISION); 
            BGMManager.Instance.Play(BGMPath.GARAGE_BAND_15TH_SGF3); 
        }
        else{  
            SEManager.Instance.Play(SEPath.SGF_MSE_RETURN); 
        } 
        //　コルーチンを開始 
        StartCoroutine("LoadData"); 
    }
            
    IEnumerator LoadData() { // シーンの読み込みをする
        async=SceneManager.LoadSceneAsync(m_scene); //　読み込みが終わるまで進捗状況をスライダーの値に反映させる 
        while (!async.isDone) { 
            var progressVal=Mathf.Clamp01(async.progress / 0.9f); 
            slider.value=progressVal;
            yield return null; 
        } 
    }  
}

LoadingSceneのスクリプトも、今回かなり力を入れたスクリプトで、主に画面遷移時に使用します。

Start()処理で念の為、シーン遷移時に削除予定のオブジェクトを全てアクティブにしています。そして、Update()で特定のキーが押されたときに関数が動くようにしています。

NextScene()では、ロードに使う画面とUIを表示し、サウンドマネージャーから音を鳴らし、子ルーチンを開始します。サウンドマネージャーも作ったけど、ファイルが多いので今回は割愛します。

LoadData()では、次のシーンに遷移しつつその状況をスライドバーにて表現しています。

以上のような一連の流れでこのゲームはシーン遷移を行います。今までのゲームではあまり気にせず遷移させていたのである意味で思い出深いスクリプトです。
